Create database
""
create database `Hermes`;

use `Hermes`;

USE `Hermes`;
CREATE TABLE `Hermes`.`users`(
    `ID` INT NOT NULL AUTO_INCREMENT,
    `Name` VARCHAR(50) NOT NULL,
    `Email` VARCHAR(50) NOT NULL,
    `Password` VARCHAR(50),
    `Created` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY(`ID`)
);

CREATE TABLE `Hermes`.`journals`(
    `ID` INT NOT NULL AUTO_INCREMENT,
    `Name` VARCHAR(50) NOT NULL,
    `Email` VARCHAR(50) NOT NULL,
    `Password` VARCHAR(50),
    `Journal` VARCHAR(20) NOT NULL,
    `Karma` VARCHAR(20) NOT NULL,
    `Created` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY(`ID`)
);

CREATE TABLE `Hermes`.`posts`(
    `PostID` INT NOT NULL AUTO_INCREMENT,
    `JournalID` INT not null,
    `Created` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY(`PostID`),
    FOREIGN KEY(`JournalID`) REFERENCES `Hermes`.`journals`(`ID`)
)

CREATE TABLE `Hermes`.`comments`(
    `CommentID` INT NOT NULL AUTO_INCREMENT,
    `CommentOwnerID` INT not null,
    `Created` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY(`CommentID`),
    FOREIGN KEY(`CommentOwnerID`) REFERENCES `Hermes`.`users`(`ID`)
)
""
